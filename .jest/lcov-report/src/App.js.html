import {

  Switch,
  Route

import WebFont from 'webfontloader';

import Footer from './components/footer';
 
import Home from './views/home';
import Wedding from './views/wedding';
import Registry from './views/registry';

  google: {

        </Switch>
      <Footer

        builtBy="The Groom"
      />
    </div>
